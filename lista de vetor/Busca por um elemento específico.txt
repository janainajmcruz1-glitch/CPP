#include <iostream>
using namespace std;

int main() {
    int vetor[10];
    cout << "Digite 10 números:" << endl;
    for(int i = 0; i < 10; i++) {
        cin >> vetor[i];
    }

    int elemento;
    cout << "Digite o número que deseja buscar: ";
    cin >> elemento;

    bool encontrado = false;
    for(int i = 0; i < 10; i++) {
        if(vetor[i] == elemento) {
            encontrado = true;
            break;
        }
    }

    if(encontrado)
        cout << "O número está presente no vetor." << endl;
    else
        cout << "O número NÃO está presente no vetor." << endl;

    return 0;
}
