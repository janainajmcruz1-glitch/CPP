#include <iostream>
using namespace std;

int main() {
    int vetor1[5], vetor2[5];
    cout << "Digite 5 elementos do primeiro vetor:" << endl;
    for(int i = 0; i < 5; i++) {
        cin >> vetor1[i];
    }

    cout << "Digite 5 elementos do segundo vetor:" << endl;
    for(int i = 0; i < 5; i++) {
        cin >> vetor2[i];
    }

    int produtoEscalar = 0;
    for(int i = 0; i < 5; i++) {
        produtoEscalar += vetor1[i] * vetor2[i];
    }

    cout << "Produto escalar: " << produtoEscalar << endl;
    return 0;
}
